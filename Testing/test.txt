public class EscribirComoPersona {

    public static void escribir(String texto, int delayMilis) {
        for (char c : texto.toCharArray()) {
            System.out.print(c);       // Imprime un carácter
            System.out.flush();        // Lo muestra inmediatamente
            try {
                Thread.sleep(delayMilis); // Espera un poco antes del siguiente
            } catch (InterruptedException e) {
                Thread.currentThread().interrupt();
            }
        }
        System.out.println(); // Al final, salto de línea
    }

    public static void main(String[] args) {
        escribir("Iniciando aplicación... Cargando menú de opciones:", 100);
        escribir("Iniciando aplicación... Cargando menú de opciones:", 85);
        escribir("Iniciando aplicación... Cargando menú de opciones:", 90);
    }
}
